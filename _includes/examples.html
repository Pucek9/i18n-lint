Examples

CLI

Lint a single HTML file

{% highlight bash %} $ hslint index.html {% endhighlight %}

Lint a single EJS file

{% highlight bash %} $ hslint -t "<%,%>" index.ejs $ hslint --template-delimiters "<%,%>" index.ejs {% endhighlight %}

Lint multiple files

{% highlight bash %} $ hslint *.html $ hslint somewhere/**/*.html {% endhighlight %}

Check 'content', 'title' and 'alt' HTML attributes

{% highlight bash %} $ hslint -a "content,title,alt" *.html $ hslint --attributes "content,title,alt" *.html {% endhighlight %}

Ignore all 'script', 'style' and 'meta' tags

{% highlight bash %} $ hslint -i "script,style,meta" *.html $ hslint --ignore-tags "script,style,meta" *.html {% endhighlight %}

API

Lint a single HTML file

{% highlight javascript %} var errors = hslint('myfile.html'); {% endhighlight %}

Lint a single EJS file

{% highlight javascript %} var errors = hslint('myfile.ejs', { templateDelimiters: ['<%', '%>'] }); {% endhighlight %}

Lint multiple files

{% highlight javascript %} var errors = hslint('somewhere/**/*.html'); {% endhighlight %}

Check 'content', 'title' and 'alt' HTML attributes

{% highlight javascript %} var errors = hslint('myfile.ejs', { attributes: ['content', 'title', 'alt'] }); {% endhighlight %}

Ignore all 'script', 'style' and 'meta' tags

{% highlight javascript %} var errors = hslint('myfile.ejs', { ignoreTags: ['script', 'style', 'meta'] }); {% endhighlight %}
