Examples

CLI

Lint a single HTML file

{% highlight bash %} $ i18n-lint index.html {% endhighlight %}

Lint a single EJS file

{% highlight bash %} $ i18n-lint -t "<%,%>" index.ejs $ i18n-lint --template-delimiters "<%,%>" index.ejs {% endhighlight %}

Lint multiple files

{% highlight bash %} $ i18n-lint *.html $ i18n-lint somewhere/**/*.html {% endhighlight %}

Check 'content', 'title' and 'alt' HTML attributes

{% highlight bash %} $ i18n-lint -a "content,title,alt" *.html $ i18n-lint --attributes "content,title,alt" *.html {% endhighlight %}

Ignore all 'script', 'style' and 'meta' tags

{% highlight bash %} $ i18n-lint -i "script,style,meta" *.html $ i18n-lint --ignore-tags "script,style,meta" *.html {% endhighlight %}

API

Lint a single HTML file

{% highlight javascript %} var errors = I18nLint('myfile.html'); {% endhighlight %}

Lint a single EJS file

{% highlight javascript %} var errors = I18nLint('myfile.ejs', { templateDelimiters: ['<%', '%>'] }); {% endhighlight %}

Lint multiple files

{% highlight javascript %} var errors = I18nLint('somewhere/**/*.html'); {% endhighlight %}

Check 'content', 'title' and 'alt' HTML attributes

{% highlight javascript %} var errors = I18nLint('myfile.ejs', { attributes: ['content', 'title', 'alt'] }); {% endhighlight %}

Ignore all 'script', 'style' and 'meta' tags

{% highlight javascript %} var errors = I18nLint('myfile.ejs', { ignoreTags: ['script', 'style', 'meta'] }); {% endhighlight %}
